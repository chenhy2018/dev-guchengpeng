# ffmpeg static library generate manifest file
export FFMPEG_RELEASE_TAG=n3.4.4
git clone -b ${FFMPEG_RELEASE_TAG} https://github.com/ffmpeg/ffmpeg.git /var/local/git/ffmpeg && \
        echo "--- installing ffmpeg ---" && \
cd /var/local/git/ffmpeg && \
    ./configure \
        --enable-gpl \
        --enable-version3 \
        --enable-nonfree \
        --disable-shared \
        --enable-static \
        --enable-small \
        --disable-doc \
        --disable-programs \
        --disable-debug \
        \
        --disable-avdevice \
        --enable-avcodec \
        --enable-avformat \
        --enable-swresample \
        --enable-swscale \
        --disable-postproc \
        --enable-avfilter \
        --disable-avresample \
        \
        --disable-encoders \
        --disable-decoders \
        --disable-hwaccels \
        --enable-muxers \
        --enable-demuxers \
        --disable-parsers \
        --enable-bsfs \
        --enable-protocols \
        --disable-devices \
        --disable-filters \
        \
        --disable-cuda \
        --disable-cuvid \
        --disable-nvenc \
        --disable-v4l2_m2m && \
        make && make install && make clean
